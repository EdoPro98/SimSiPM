27  std::vector<int32_t> out(v.size());
28 
29  const double qlevels = pow(2,nbits);

31  const double width = range / gainlinear / qlevels;
32

45 std::vector<double> SiPMAdc::addJitter(std::vector<double>& signal, const double jit) const {
46  const uint32_t n = signal.size();
47  std::vector<double> lsignalshift = signal; // Copy of signal
48  double jitweight;
49 
50  // Right shift (jitter is positive -> need to move signal "to right")
51  if (jit > 0) {
52  const uint32_t jitidx = floor(jit);
53  jitweight = jit - jitidx;
54  std::rotate(signal.rbegin(), signal.rbegin() + jitidx, signal.rend());
55  // Signal shifted by 1 idx to right
56  std::rotate(lsignalshift.rbegin(), lsignalshift.rbegin() + 1,
57  lsignalshift.rend());
58  }
59  // Left shift (jitter is negative -> need to move signal "to left")
60  else {
61  const uint32_t jitidx = floor(-jit);
62  jitweight = -jit - jitidx;
63  std::rotate(signal.begin(), signal.begin() + jitidx, signal.end());
64  // Signal shifted by 1 to left
65  std::rotate(lsignalshift.begin(), lsignalshift.begin() + 1,
66  lsignalshift.end());
67  }
68 
69  // In this part i take into account of a signal shift smaller thant 1 idx.
70  // In this case the resulting signal is the wighted sum of the original signal
71  // and the signal shiftedby one.
72 #ifdef __AVX2__
73  const uint32_t last = n - n % 4;
74  const __m256d __jitweight = _mm256_set1_pd(jitweight);
75  __m256d __signal, __signalshift;
76  for (uint32_t i = 0; i < last; i += 4) {
77  __signal = _mm256_loadu_pd(&signal[i]);
78  __signalshift = _mm256_loadu_pd(&lsignalshift[i]);
79  __signal = _mm256_fmadd_pd(
80  __jitweight, _mm256_sub_pd(__signalshift, __signal), __signal);
81  _mm256_storeu_pd(&signal[i], __signal);
82  }
83  for (uint32_t i = last; i < n; ++i) {
84  signal[i] += jitweight * (lsignalshift[i] - signal[i]);
85  }
86 #else
87  for (uint32_t i = 0; i < n; ++i) {
88  // (1 - weigth) * lsignal + weigth * lsignalshift
89  signal[i] += jitweight * (lsignalshift[i] - signal[i]);
90  }
91 #endif
92  return signal;
93 }
