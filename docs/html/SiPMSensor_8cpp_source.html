69 const std::vector<double> SiPMSensor::signalShape() const {
70  const uint32_t nSignalPoints = m_Properties.nSignalPoints();
71  const double sampling = m_Properties.sampling();
72  const double tr = m_Properties.risingTime() / sampling;
73  const double tff = m_Properties.fallingTimeFast() / sampling;
74  std::vector<double> lSignalShape(nSignalPoints);
75 
76  if (m_Properties.hasSlowComponent()) {
77  const double tfs = m_Properties.fallingTimeSlow() / sampling;
78  const double slf = m_Properties.slowComponentFraction();
79 
80  for (int32_t i = 0; i < nSignalPoints; ++i) {
81  lSignalShape[i] =
82  (1 - slf) * exp(-i / tff) + slf * exp(-i / tfs) - exp(-i / tr);
83  }
84  } else {
85  for (int32_t i = 0; i < nSignalPoints; ++i) {
86  lSignalShape[i] = exp(-i / tff) - exp(-i / tr);
87  }
88  }
89 
90  const double peak =
91  *std::max_element(lSignalShape.begin(), lSignalShape.end());
92 
93  for (uint32_t i = 0; i < nSignalPoints; ++i) { lSignalShape[i] /= peak; }
94 
95  return lSignalShape;
96 }

98 const double SiPMSensor::evaluatePde(const double aPhotonWavelength) const {
99  const std::map<double, double> pde = m_Properties.pdeSpectrum();
100  auto right = pde.upper_bound(aPhotonWavelength);
101  if (right == pde.end()) { --right; }
102  if (right == pde.begin()) { ++right; }
103  auto left = right;
104  --left;
105  const double delta =
106  (aPhotonWavelength - left->first) / (right->first - left->first);
107  return delta * right->second + (1 - delta) * left->second;
108 }

143 void SiPMSensor::addDcrEvents() {
144  const double signalLength = m_Properties.signalLength();
145  const double meanDcr = 1e9 / m_Properties.dcr();
146  const int32_t nSideCells = m_Properties.nSideCells() - 1;
147 
148  double last = -100;
149 
150  while (last < signalLength) {
151  last += m_rng.randExponential(meanDcr);
152  if ((last > 0) && (last < signalLength)) {
153  int32_t row = m_rng.randInteger(nSideCells);
154  int32_t col = m_rng.randInteger(nSideCells);
155 
156  m_Hits.emplace_back(last, 1, row, col, SiPMHit::HitType::kDarkCount);
157 
158  m_nTotalHits++;
159  m_nDcr++;
160  }

164 void SiPMSensor::addPhotoelectrons() {
165  const uint32_t nPhotons = m_PhotonTimes.size();
166  const double pde = m_Properties.pde();
167  m_Hits.reserve(nPhotons);
168 
169  switch (m_Properties.pdeType()) {
170  case (SiPMProperties::PdeType::kNoPde):
171  // Add all photons
172  for (uint32_t i = 0; i < nPhotons; ++i) {
173  std::pair<int32_t, int32_t> position = hitCell();
174 
175  m_Hits.emplace_back(m_PhotonTimes[i], 1, position.first, position.second,
176  SiPMHit::HitType::kPhotoelectron);
177 
178  m_nTotalHits++;
179  m_nPe++;
180  }
181  break;
182 
183  case (SiPMProperties::PdeType::kSimplePde):
184  // Simple pde
185  for (uint32_t i = 0; i < nPhotons; ++i) {
186  if (isDetected(pde)) {
187  std::pair<int32_t, int32_t> position = hitCell();
188 
189  m_Hits.emplace_back(m_PhotonTimes[i], 1, position.first,
190  position.second, SiPMHit::HitType::kPhotoelectron);
191 
192  m_nTotalHits++;
193  m_nPe++;
194  }
195  }
196  break;
197 
198  case (SiPMProperties::PdeType::kSpectrumPde):
199  // Evaluate pde based on wavelength
200  for (uint32_t i = 0; i < nPhotons; ++i) {
201  if (isDetected(evaluatePde(m_PhotonWavelengths[i]))) {
202  std::pair<int32_t, int32_t> position = hitCell();
203 
204  m_Hits.emplace_back(m_PhotonTimes[i], 1, position.first,
205  position.second, SiPMHit::HitType::kPhotoelectron);
206 
207  m_nTotalHits++;
208  m_nPe++;
209  }
210  }
211  break;
212  } /* SWITCH */
213 }

215 void SiPMSensor::addXtEvents() {
216  const double xt = m_Properties.xt();
217 
218  uint32_t currentCellIdx = 0;
219  while (currentCellIdx < m_nTotalHits) {
220  SiPMHit* hit = &m_Hits[currentCellIdx];
221  currentCellIdx++;
222  uint32_t nxt = m_rng.randPoisson(xt);
223 
224  if (nxt == 0) { continue; }
225  int32_t xtGeneratorRow = hit->row();
226  int32_t xtGeneratorCol = hit->col();
227  double xtTime = hit->time();
228 
229  for (uint32_t j = 0; j < nxt; ++j) {
230  int32_t rowAdd, colAdd;
231  do {
232  rowAdd = m_rng.randInteger(2) - 1;
233  colAdd = m_rng.randInteger(2) - 1;
234  } while (rowAdd + colAdd == 0);
235  int32_t xtRow = xtGeneratorRow + rowAdd;
236  int32_t xtCol = xtGeneratorCol + colAdd;
237 
238  if (isInSensor(xtRow, xtCol)) {
239  m_Hits.emplace_back(xtTime, 1, xtRow, xtCol,
240  SiPMHit::HitType::kOpticalCrosstalk);
241  ++m_nTotalHits;
242  ++m_nXt;
243  }
244  } /* FOR XT */
245  } /* WHILE HIT */
246 }

248 void SiPMSensor::addApEvents() {
249  const double ap = m_Properties.ap();
250  const double tauApFast = m_Properties.tauApFast();
251  const double tauApSlow = m_Properties.tauApSlow();
252  const double signalLength = m_Properties.signalLength();
253  const double recoveryTime = m_Properties.recoveryTime();
254  const double slowFraction = m_Properties.apSlowFraction();
255 
256  // Cannot use iterator based loop becouse of reallocation
257  uint32_t currentCellIdx = 0;
258  while (currentCellIdx < m_nTotalHits) {
259  SiPMHit* hit = &m_Hits[currentCellIdx];
260  currentCellIdx++;
261 
262  uint32_t nap = m_rng.randPoisson(ap);
263 
264  if (nap == 0) { continue; }
265  double apGeneratorTime = hit->time();
266 
267  // Choose fast or slow ap
268  for (uint32_t j = 0; j < nap; ++j) {
269  double apDelay;
270  if (m_rng.Rand() < slowFraction) {
271  apDelay = m_rng.randExponential(tauApSlow);
272  } else {
273  apDelay = m_rng.randExponential(tauApFast);
274  }
275 
276  // If ap event is in signal window
277  if (apGeneratorTime + apDelay < signalLength) {
278  double apAmplitude =
279  hit->amplitude() * (1 - exp(-apDelay / recoveryTime));
280 
281  m_Hits.emplace_back(apGeneratorTime + apDelay, apAmplitude, hit->row(),
282  hit->col(), SiPMHit::HitType::kAfterPulse);
283 
284  m_nTotalHits++;
285  m_nAp++;
286  }
287  } /* FOR AP */
288  } /* WHILE HIT */
289 }

291 void SiPMSensor::calculateSignalAmplitudes() {
292  sortHits();
293  const std::vector<uint32_t> cellId = getCellIds();
294  const std::unordered_set<uint32_t> uniqueCellId(cellId.begin(), cellId.end());
295  const double cellRecovery = m_Properties.recoveryTime();
296 
297  for (auto itr = uniqueCellId.begin(); itr != uniqueCellId.end(); ++itr) {
298  // If cell hitted more than once
299  if (std::count(cellId.begin(), cellId.end(), *itr) > 1) {
300  double previousHitTime = 0;
301  for (auto hit = m_Hits.begin(); hit != m_Hits.end(); ++hit) {
302  if (hit->id() == *itr) {
303  if (previousHitTime == 0) {
304  previousHitTime = hit->time();
305  } else {
306  double delay = hit->time() - previousHitTime;
307  hit->amplitude() = 1 - exp(-delay / cellRecovery);
308  previousHitTime = hit->time();
309  }
310  }
311  }
312  }
313  }

316 void SiPMSensor::generateSignal() {
317  const uint32_t nSignalPoints = m_Properties.nSignalPoints();
318  const double sampling = m_Properties.sampling();
319 
320  m_Signal = m_rng.randGaussian(0, m_Properties.snrLinear(), nSignalPoints);
321 
322  for (auto hit = m_Hits.begin(); hit != m_Hits.end(); ++hit) {
323  const int32_t time = hit->time() / sampling;
324  const double amplitude =
325  hit->amplitude() * m_rng.randGaussian(1, m_Properties.ccgv());
326 
327 #ifdef __AVX2__
328  uint32_t i;
329  const __m256d __amplitude = _mm256_set1_pd(amplitude);
330  for (i = time; i < nSignalPoints - 4; i += 4) {
331  __m256d __signal = _mm256_loadu_pd(&m_Signal[i]); // Load s
332  __m256d __shape = _mm256_loadu_pd(&m_SignalShape[i - time]); // Load sh
333  __signal = _mm256_fmadd_pd(__shape, __amplitude, __signal); // s += sh * a
334  _mm256_storeu_pd(&m_Signal[i], __signal); // Save signal
335  }
336  while (i < nSignalPoints) {
337  m_Signal[i] += m_SignalShape[i - time] * amplitude;
338  ++i;
339  }
340 #else
341  for (uint32_t i = time; i < nSignalPoints; ++i) {
342  m_Signal[i] += m_SignalShape[i - time] * amplitude;
343  }
344 #endif
345  }
346 }
