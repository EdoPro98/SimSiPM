63 const std::vector<double> SiPMSensor::signalShape() const {
64  const uint32_t nSignalPoints = m_Properties.nSignalPoints();
65  const double sampling = m_Properties.sampling();
66  const double tr = m_Properties.risingTime() / sampling;
67  const double tff = m_Properties.fallingTimeFast() / sampling;
68  std::vector<double> lSignalShape(nSignalPoints);
69 
70  if (m_Properties.hasSlowComponent()) {
71  const double tfs = m_Properties.fallingTimeSlow() / sampling;
72  const double slf = m_Properties.slowComponentFraction();
73 
74  for (int32_t i = 0; i < nSignalPoints; ++i) {
75  lSignalShape[i] =
76  (1 - slf) * exp(-i / tff) + slf * exp(-i / tfs) - exp(-i / tr);
77  }
78  } else {
79  for (int32_t i = 0; i < nSignalPoints; ++i) {
80  lSignalShape[i] = exp(-i / tff) - exp(-i / tr);
81  }
82  }
83 
84  const double peak =
85  *std::max_element(lSignalShape.begin(), lSignalShape.end());
86 
87  for (uint32_t i = 0; i < nSignalPoints; ++i) { lSignalShape[i] /= peak; }
88 
89  return lSignalShape;
90 }

92 const double SiPMSensor::evaluatePde(const double aPhotonWavelength) const {
93  const std::map<double, double> pde = m_Properties.pdeSpectrum();
94  auto right = pde.upper_bound(aPhotonWavelength);
95  if (right == pde.end()) { --right; }
96  if (right == pde.begin()) { ++right; }
97  auto left = right;
98  --left;
99  const double delta =
100  (aPhotonWavelength - left->first) / (right->first - left->first);
101  return delta * right->second + (1 - delta) * left->second;
102 }

316  const double sampling = m_Properties.sampling();
317 
318  m_Signal = m_rng.randGaussian(0, m_Properties.snrLinear(), nSignalPoints);
319 
320  for (auto hit = m_Hits.begin(); hit != m_Hits.end(); ++hit) {
321  const int32_t time = hit->time() / sampling;
322  const double amplitude =

324 
325 #ifdef __AVX2__
326  uint32_t i;
327  const __m256d __amplitude = _mm256_set1_pd(amplitude);
328  for (i = time; i < nSignalPoints - 4; i += 4) {
329  __m256d __signal = _mm256_loadu_pd(&m_Signal[i]); // Load s
330  __m256d __shape = _mm256_loadu_pd(&m_SignalShape[i - time]); // Load sh
331  __signal =
332  _mm256_fmadd_pd(__shape, __amplitude, __signal); // s += sh * a
333  _mm256_storeu_pd(&m_Signal[i], __signal); // Save signal
334  }
335  while (i < nSignalPoints) {
336  m_Signal[i] += m_SignalShape[i - time] * amplitude;
337  ++i;
338  }
339 #else
340  for (uint32_t i = time; i < nSignalPoints; ++i) {
341  m_Signal[i] += m_SignalShape[i - time] * amplitude;
342  }
343 #endif
344  }
